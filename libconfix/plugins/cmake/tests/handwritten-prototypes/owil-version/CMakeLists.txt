PROJECT(test)
CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

FIND_PROGRAM(OWIL_CONFIG owil-config)

IF (NOT OWIL_CONFIG)
    MESSAGE(STATUS "OWIL: owil-config not found; leaving version unknown")
    SET(OWIL_VERSION unknown)
ELSE (NOT OWIL_CONFIG)
    EXECUTE_PROCESS(
        COMMAND ${OWIL_CONFIG} --version
        OUTPUT_VARIABLE OWIL_VERSION
        OUTPUT_STRIP_TRAILING_WHITESPACE
        )
ENDIF (NOT OWIL_CONFIG)

MESSAGE(STATUS "OWIL_VERSION: ${OWIL_VERSION}")
