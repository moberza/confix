ax_am_macros

Synopsis

AX_AM_MACROS

Description

Adds support for macros that create automake rules. You must manually add the following line 

  @INC_AMINCLUDE@

to your Makefile.am files.

Author

Tom Howard <tomhoward@users.sf.net>

Last Modified

2005-01-14

M4 Source Code 

AC_DEFUN([AX_AM_MACROS],
[
AC_MSG_NOTICE([adding automake macro support])
AMINCLUDE="aminclude.am"
AC_SUBST(AMINCLUDE)
AC_MSG_NOTICE([creating $AMINCLUDE])
AMINCLUDE_TIME=`date`
AX_PRINT_TO_FILE([$AMINCLUDE],[[
# generated automatically by configure from AX_AUTOMAKE_MACROS
# on $AMINCLUDE_TIME

]])

INC_AMINCLUDE="include \$(top_builddir)/$AMINCLUDE"
AC_SUBST(INC_AMINCLUDE)
])
