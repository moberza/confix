ac_func_mkdir

Synopsis

AC_FUNC_MKDIR

Description

Check whether mkdir() is mkdir or _mkdir, and whether it takes one or two arguments.

This macro can define HAVE_MKDIR, HAVE__MKDIR, and MKDIR_TAKES_ONE_ARG, which are expected to be used as follows: 

  #if HAVE_MKDIR
  #  if MKDIR_TAKES_ONE_ARG
       /* MinGW32 */
  #    define mkdir(a, b) mkdir(a)
  #  endif
  #else
  #  if HAVE__MKDIR
       /* plain Windows 32 */
  #    define mkdir(a, b) _mkdir(a)
  #  else
  #    error "Don't know how to create a directory on this system."
  #  endif
  #endif

M4 Source Code 

AC_DEFUN([AC_FUNC_MKDIR],
[AC_CHECK_FUNCS([mkdir _mkdir])
AC_CACHE_CHECK([whether mkdir takes one argument],
               [ac_cv_mkdir_takes_one_arg],
[AC_TRY_COMPILE([
#include <sys/stat.h>
#if HAVE_UNISTD_H
#  include <unistd.h>
#endif
], [mkdir (".");],
[ac_cv_mkdir_takes_one_arg=yes], [ac_cv_mkdir_takes_one_arg=no])])
if test x"$ac_cv_mkdir_takes_one_arg" = xyes; then
  AC_DEFINE([MKDIR_TAKES_ONE_ARG], 1,
            [Define if mkdir takes only one argument.])
fi
])
